        async function connectWebSocket(id) {
            const conn = wsConnections.find(c => c.id === id);
            if (!conn) {
                appendLog(`Соединение с ID ${id} не найдено.`, 'error');
                return;
            }

            if (conn.wsInstance && (conn.wsInstance.readyState === WebSocket.OPEN || conn.wsInstance.readyState === WebSocket.CONNECTING)) {
                appendLog(`Соединение ${conn.name} уже подключено или подключается.`, 'info', id);
                return;
            }

            const wsUrl = document.getElementById(`ws-url-${id}`).value.trim();
            if (!wsUrl) {
                appendLog(`Пожалуйста, введите адрес WebSocket сервера для ${conn.name}.`, 'error', id);
                return;
            }

            appendLog(`Попытка подключения к WebSocket: ${wsUrl} (${conn.name})...`, 'info', id);
            updateWsStatusDisplay(id, 'Connecting...');
            conn.status = 'Connecting...';
            updateGlobalConnectionStatus();

            conn.wsInstance = new WebSocket(wsUrl);

            conn.wsInstance.onopen = () => {
                appendLog(`WebSocket ${conn.name} подключен.`, 'info', id);
                updateWsStatusDisplay(id, 'Connected');
                conn.status = 'Connected';
                updateGlobalConnectionStatus();
                // Send auth token immediately after connection
                if (authToken) {
                    conn.wsInstance.send(JSON.stringify({ type: 'auth', token: authToken }));
                    appendLog(`Отправлен токен авторизации для ${conn.name}.`, 'info', id);
                } else {
                    appendLog(`Нет токена авторизации для ${conn.name}.`, 'error', id);
                }

                // Если это первое подключение или активное соединение, обновить статус AMI
                if (activeWsId === id || activeWsId === null) {
                    updateGlobalAmiStatus('Подключено');
                    activeWsId = id; // Сделать это активным, если еще нет активного
                    updateActiveWsSelect();
                }
            };

            conn.wsInstance.onmessage = event => {
                const message = JSON.parse(event.data);
                switch (message.type) {
                    case 'server_status':
                        appendLog(`Сервер (${conn.name}): ${message.message}`, 'info', id);
                        break;
                    case 'ami_status':
                        conn.amiStatus = message.status;
                        if (activeWsId === id) { // Обновляем глобальный статус AMI только для активного соединения
                            updateGlobalAmiStatus(message.status);
                        }
                        appendLog(`AMI статус (${conn.name}): ${message.status}`, 'info', id);
                        break;
                    case 'ami_event':
                        appendLog(`AMI Событие (${conn.name}): ${JSON.stringify(message.data)}`, 'ami_event', id);
                        // Handle call status events for conference monitoring
                        if (message.data.Event === 'Newchannel' || message.data.Event === 'Hangup' ||
                            message.data.Event === 'ConfbridgeJoin' || message.data.Event === 'ConfbridgeLeave') {
                            handleConferenceCallEvent(message.data, id);
                        }
                        break;
                    case 'ami_response':
                        appendLog(`AMI Ответ (${conn.name}): ${JSON.stringify(message.data)}`, 'ami_response', id);
                        break;
                    case 'ami_action_response':
                        appendLog(`Ответ на действие "${message.action}" (${conn.name}): ${JSON.stringify(message.data)}`, 'ami_response', id);
                        break;
                    case 'conf_originate_response':
                        appendLog(`Ответ на присоединение к конференции (${conn.name}): ${JSON.stringify(message.data)}`, 'ami_response', id);
                        break;
                    case 'dtmf_digit':
                        // DTMF всегда относится к активной конференции, независимо от WS
                        accumulatedDtmf += message.digit;
                        dtmfDisplayDiv.textContent = accumulatedDtmf || 'Ожидание DTMF...';
                        appendLog(`DTMF "${message.digit}" получено в конференции "${message.conference}" (через ${conn.name})`, 'dtmf', id);
                        break;
                    case 'confbridge_event':
                        appendLog(`Событие Конференции (${conn.name}): ${JSON.stringify(message.data)}`, 'ami_event', id);
                        break;
                    case 'ami_error':
                        appendLog(`Ошибка AMI (${conn.name}): ${message.message}`, 'error', id);
                        break;
                    case 'error':
                        appendLog(`Ошибка: ${message.message}`, 'error', id);
                        break;
                    case 'adb_device_update':
                        populateAdbDevices(message.devices);
                        appendLog(`Обновлен список ADB устройств: ${message.devices.length} найдено.`, 'adb', id);
                        break;
                    case 'adb_action_response':
                        if (message.success) {
                            appendLog(`ADB: Устройство ${message.deviceId}, действие "${message.actionType}" успешно. Вывод: "${message.output}"`, 'adb', id);
                        } else {
                            appendLog(`ADB: Устройство ${message.deviceId}, действие "${message.actionType}" ошибка: ${message.error}`, 'error', id);
                        }
                        break;
                    case 'auth_status':
                        if (message.success) {
                            appendLog(`Авторизация WebSocket для ${conn.name} успешна.`, 'info', id);
                        } else {
                            appendLog(`Авторизация WebSocket для ${conn.name} не удалась: ${message.message}`, 'error', id);
                            // Optionally disconnect if auth fails
                            // conn.wsInstance.close();
                        }
                        break;
                    default:
                        appendLog(`Неизвестное сообщение (${conn.name}): ${JSON.stringify(message)}`, 'info', id);
                }
            };

            conn.wsInstance.onclose = () => {
                appendLog(`WebSocket ${conn.name} отключен. Попытка переподключения через 3 секунды...`, 'error', id);
                updateWsStatusDisplay(id, 'Disconnected');
                conn.status = 'Disconnected';
                conn.amiStatus = 'Unknown';
                updateGlobalConnectionStatus();
                if (activeWsId === id) {
                    updateGlobalAmiStatus('Отключено');
                }
                setTimeout(() => connectWebSocket(id), 3000); // Попытка переподключения
            };

            conn.wsInstance.onerror = error => {
                appendLog(`WebSocket ошибка ${conn.name}: ${error.message}`, 'error', id);
                console.error(`WebSocket Error (${conn.name}):`, error);
                conn.wsInstance.close(); // Закрываем соединение, чтобы сработал onclose и попытка переподключения
            };
        }

        function disconnectWebSocket(id) {
            const conn = wsConnections.find(c => c.id === id);
            if (conn && conn.wsInstance && conn.wsInstance.readyState === WebSocket.OPEN) {
                conn.wsInstance.close();
                appendLog(`WebSocket ${conn.name} отключен вручную.`, 'info', id);
            } else {
                appendLog(`WebSocket ${conn.name} не активен или уже отключен.`, 'info', id);
            }
        }

        function getActiveWs() {
            if (activeWsId === null) {
                appendLog('Нет активного WebSocket соединения для отправки команд.', 'error');
                return null;
            }
            const activeConn = wsConnections.find(conn => conn.id === activeWsId);
            if (!activeConn || !activeConn.wsInstance || activeConn.wsInstance.readyState !== WebSocket.OPEN) {
                appendLog(`Активное соединение (ID: ${activeWsId}) не активно. Пожалуйста, подключитесь.`, 'error');
                return null;
            }
            return activeConn.wsInstance;
        }

        // --- Persistence (localStorage) ---
        function saveConnectionsToLocalStorage() {
            localStorage.setItem('wsConnections', JSON.stringify(wsConnections.map(conn => ({
                id: conn.id,
                name: conn.name,
                url: conn.url,
                sipAccounts: conn.sipAccounts, // Save SIP accounts
                defaultConference: conn.defaultConference // Save default conference
            }))));
        }

        function loadConnectionsFromLocalStorage() {
            const storedConnections = localStorage.getItem('wsConnections');
            if (storedConnections) {
                const parsedConnections = JSON.parse(storedConnections);
                wsConnections = parsedConnections.map(connData => ({
                    ...connData,
                    wsInstance: null, // WebSocket instance will be re-created on connect
                    status: 'Disconnected',
                    amiStatus: 'Unknown',
                    // Ensure defaultConference exists, even if not in old stored data
                    defaultConference: connData.defaultConference || '' 
                }));
                // Restore wsCounter to ensure unique IDs for new connections
                wsCounter = wsConnections.length > 0 ? Math.max(...wsConnections.map(c => c.id)) + 1 : 0;
            } else {
                // Add default connections if no stored data
                addWsConnection('Сервер 1 (Local)', `ws://localhost:${WS_PORT}`, [], 'myconference'); // Default conference for demo
                addWsConnection('Сервер 2 (LAN)', `ws://192.168.1.100:${WS_PORT}`, [], 'myconference');
                addWsConnection('Сервер 3 (Remote)', `ws://your-remote-server-ip:${WS_PORT}`, [], 'myconference');
            }
        }

        function handleConferenceCallEvent(eventData, wsId) {
            let message = '';
            let type = 'info';
            const conference = eventData.Conference || 'Unknown Conference';
            const channel = eventData.Channel || 'Unknown Channel';
            const uniqueid = eventData.Uniqueid || `simulated-${Date.now()}`; // Generate unique ID for simulated calls
            const connectedlinenum = eventData.ConnectedLineNum || eventData.CallerIDNum || 'N/A';
            const connectedlinename = eventData.ConnectedLineName || 'N/A';

            switch (eventData.Event) {
                case 'Newchannel':
                    message = `Новый канал создан: ${channel} (ID: ${uniqueid}) для ${connectedlinename} (${connectedlinenum})`;
                    type = 'start';
                    break;
                case 'Hangup':
                    message = `Канал завершен: ${channel} (ID: ${uniqueid})`;
                    type = 'end';
                    break;
                case 'ConfbridgeJoin':
                    message = `Участник ${channel} присоединился к конференции "${conference}"`;
                    type = 'start';
                    break;
                case 'ConfbridgeLeave':
                    message = `Участник ${channel} покинул конференцию "${conference}"`;
                    type = 'end';
                    break;
                case 'SimulatedCallStart': // Custom event for client-side simulation
                    message = `Симулирован старт звонка для номера ${connectedlinenum} (ID: ${uniqueid})`;
                    type = 'start';
                    break;
                case 'SimulatedCallEnd': // Custom event for client-side simulation
                    message = `Симулирован отбой звонка для номера ${connectedlinenum} (ID: ${uniqueid})`;
                    type = 'end';
                    break;
                default:
                    message = `Неизвестное событие конференции: ${JSON.stringify(eventData)}`;
                    type = 'info';
            }
            appendConferenceCallStatus(message, type);
            appendLog(`Событие конференции (WS:${wsId}): ${message}`, type, wsId);
        }

        loadCsvButton.addEventListener('click', () => {

        function parseCsvData(csvText) {
            csvData = [];
            const lines = csvText.split('\n').filter(line => line.trim() !== ''); // Filter empty lines
            csvDataDisplayDiv.innerHTML = '';
            automationStatusDiv.classList.add('d-none');

            if (lines.length === 0) {
                appendLog('CSV файл пуст или имеет неверный формат.', 'error');
                csvDataDisplayDiv.innerHTML = '<p class="text-muted">CSV файл пуст или имеет неверный формат.</p>';
                startAutomationButton.disabled = true;
                return;
            }

            let headerSkipped = false;
            lines.forEach((line, index) => {
                // Simple CSV parsing: split by comma, assume first line is header
                const parts = line.split(',').map(part => part.trim());
                if (parts.length === 2) {
                    if (!headerSkipped && (parts[0].toLowerCase() === 'number' || parts[0].toLowerCase() === 'номер')) {
                        // Assume first line is header and skip it
                        headerSkipped = true;
                        return;
                    }
                    csvData.push({ number: parts[0], url: parts[1] });
                    const p = document.createElement('p');
                    p.textContent = `Номер: ${parts[0]}, URL: ${parts[1]}`;
                    csvDataDisplayDiv.appendChild(p);
                } else {
                    appendLog(`Строка CSV ${index + 1} имеет неверный формат: "${line}" (ожидается "номер,URL")`, 'warning');
                }
            });

            if (csvData.length > 0) {
                appendLog(`Загружено ${csvData.length} записей из CSV.`, 'info');
                startAutomationButton.disabled = false;
                csvDataDisplayDiv.classList.remove('bg-light');
                csvDataDisplayDiv.classList.add('bg-success', 'text-white');
            } else {
                appendLog('Не удалось загрузить данные из CSV. Проверьте формат.', 'error');
                csvDataDisplayDiv.innerHTML = '<p class="text-muted">Не удалось загрузить данные из CSV. Проверьте формат.</p>';
                startAutomationButton.disabled = true;
            }
        }

        startAutomationButton.addEventListener('click', () => {
            if (csvData.length === 0) {
                appendLog('Нет данных для автоматизации. Пожалуйста, загрузите CSV файл.', 'error');
                return;
            }
            if (automationRunning) {
                appendLog('Автоматизация уже запущена.', 'info');
                return;
            }

            automationRunning = true;
            automationIndex = 0;
            startAutomationButton.disabled = true;
            stopAutomationButton.disabled = false;
            automationStatusDiv.classList.remove('d-none', 'alert-success', 'alert-danger');
            automationStatusDiv.classList.add('alert-info');
            automationStatusDiv.textContent = 'Автоматизация: Запущена. Обработка номеров...';
            appendLog('Автоматизация URL запущена.', 'info');

            // Start processing CSV entries sequentially
            processNextAutomationEntry();
        });

        function stopAutomation() {
            automationRunning = false;
            if (automationInterval) {
                clearInterval(automationInterval);
                automationInterval = null;
            }
            // Clear any pending simulated call timeouts
            activeSimulatedCalls.forEach(timeoutId => clearTimeout(timeoutId));
            activeSimulatedCalls.clear();

            startAutomationButton.disabled = false;
            stopAutomationButton.disabled = true;
            automationStatusDiv.textContent = 'Автоматизация: Остановлена.';
            automationStatusDiv.classList.remove('alert-success', 'alert-info');
            automationStatusDiv.classList.add('alert-danger');
        }

        function processNextAutomationEntry() {
            if (!automationRunning || automationIndex >= csvData.length) {
                appendLog('Автоматизация завершена: все номера обработаны.', 'info');
                automationStatusDiv.textContent = 'Автоматизация: Завершена. Все номера обработаны.';
                automationStatusDiv.classList.remove('alert-info', 'alert-danger');
                automationStatusDiv.classList.add('alert-success');
                stopAutomation();
                return;
            }

            const currentEntry = csvData[automationIndex];
            const uniqueId = `sim-${Date.now()}-${automationIndex}`; // Unique ID for simulation

            appendLog(`Автоматизация: Обработка номера ${currentEntry.number}. Открытие URL: ${currentEntry.url}`, 'info');
            automationStatusDiv.textContent = `Автоматизация: Обработка ${currentEntry.number}...`;

            // 1. Open URL
            window.open(currentEntry.url, '_blank');

            // 2. Simulate Call Start
            simulateCallStart(currentEntry.number, uniqueId);

            // 3. Schedule Call End
            const callDurationSeconds = parseInt(simulatedCallDurationInput.value) || 5;
            const timeoutId = setTimeout(() => {
                simulateCallEnd(currentEntry.number, uniqueId);
                activeSimulatedCalls.delete(uniqueId);
                automationIndex++; // Move to next entry only after call ends
                processNextAutomationEntry(); // Process next entry
            }, callDurationSeconds * 1000);
            activeSimulatedCalls.set(uniqueId, timeoutId);
        }

        // --- Функции для аудио/микрофона (заглушки) ---

        async function populateAudioDevices() {
            try {
                const stream = await navigator.mediaDevices.getUserMedia({ audio: true, video: false });
                stream.getTracks().forEach(track => track.stop());

                const devices = await navigator.mediaDevices.enumerateDevices();
                audioInputSelect.innerHTML = '<option value="">-- Выберите микрофон --</option>';
                audioOutputSelect.innerHTML = '<option value="">-- Выберите динамики --</option>';

                devices.forEach(device => {
                    const option = document.createElement('option');
                    option.value = device.deviceId;
                    option.textContent = device.label || `Устройство ${device.kind} ${device.deviceId.substring(0, 8)}...`;
                    if (device.kind === 'audioinput') {
                        audioInputSelect.appendChild(option);
                    } else if (device.kind === 'audiooutput') {
                        audioOutputSelect.appendChild(option);
                    }
                });
                appendLog('Аудиоустройства загружены.', 'info');
            } catch (err) {
                appendLog(`Ошибка загрузки аудиоустройств: ${err.name} - ${err.message}. Возможно, нет разрешений.`, 'error');
                console.error('Error enumerating devices:', err);
            }
        }

        // --- Authentication Logic ---
        async function authenticate(type) { // 'login' or 'register'
            const username = authUsernameInput.value.trim();
            const password = authPasswordInput.value.trim();

            if (!username || !password) {
                displayAuthMessage('Пожалуйста, введите имя пользователя и пароль.', 'danger');
                return;
            }

            authMessageDiv.classList.add('d-none'); // Hide previous messages

            try {
                const response = await fetch(`/${type}`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({ username, password })
                });

                const data = await response.json();

                if (response.ok) {
                    authToken = data.token;
                    currentUsername = username;
                    localStorage.setItem('authToken', authToken); // Store token
                    localStorage.setItem('username', currentUsername); // Store username
                    displayAuthMessage(data.message, 'success');
                    setTimeout(showApp, 1000); // Show main app after a short delay
                } else {
                    displayAuthMessage(data.message || 'Ошибка авторизации/регистрации', 'danger');
                }
            } catch (error) {
                console.error('Ошибка при запросе авторизации:', error);
                displayAuthMessage('Ошибка подключения к серверу авторизации.', 'danger');
            }
        }

        function displayAuthMessage(message, type) {
            authMessageDiv.textContent = message;
            authMessageDiv.classList.remove('d-none', 'alert-danger', 'alert-success');
            authMessageDiv.classList.add(`alert-${type}`);
        }

        function showAuthForm() {
            authContainer.classList.remove('d-none');
            appContainer.classList.add('d-none');
            authToken = null;
            currentUsername = null;
            localStorage.removeItem('authToken');
            localStorage.removeItem('username');
            localStorage.removeItem('wsConnections'); // Clear all stored connections on logout
            wsConnections = []; // Clear connections list
            wsCounter = 0; // Reset counter
            updateActiveWsSelect();
            updateGlobalConnectionStatus();
            updateGlobalAmiStatus('Неизвестно');
            appendLog('Вы вышли из системы.', 'info');
            stopAutomation(); // Stop any running automation on logout
        }

        function showApp() {
            authContainer.classList.add('d-none');
            appContainer.classList.remove('d-none');
            loggedInUsernameSpan.textContent = currentUsername;
            
            // Load and initialize WebSocket connections and devices after showing app
            loadConnectionsFromLocalStorage(); // Load existing connections
            updateActiveWsSelect(); // Populate dropdown and render SIP accounts for initial active connection
            
            // Connect to the first connection if available
            if (wsConnections.length > 0 && wsConnections[0].wsInstance === null) {
                activeWsId = wsConnections[0].id; // Ensure the first connection is active
                activeWsSelect.value = activeWsId;
                connectWebSocket(activeWsId);
            }

            populateAudioDevices();
            // Bootstrap's JS handles showing the first tab by default with 'show active' classes
            // No need to call showPage here explicitly unless you want to force a specific tab
        }

        // --- Event Listeners for Auth ---
        loginButton.addEventListener('click', () => authenticate('login'));
        registerButton.addEventListener('click', () => authenticate('register'));
        logoutButton.addEventListener('click', showAuthForm);


        // --- Initial Load ---
        window.onload = () => {
            if (authToken && currentUsername) {
                showApp();
            } else {
                showAuthForm();
            }
        };
